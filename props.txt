local rgbAtivo = false
local rgbConnection = nil
local propConnections = {}
local meusProps = {}

local function IsMeuProp(obj)
    return obj:IsA("Model")
        and obj.Name:find(plr.Name)
        and obj:FindFirstChild("ChangePropColor")
end

local function AddProp(prop)
    if not table.find(meusProps, prop) then
        table.insert(meusProps, prop)
    end
end

local function ScanProps()
    meusProps = {}

    local workspaceCom = workspace:FindFirstChild("WorkspaceCom")
    if not workspaceCom then return end

    for _, obj in ipairs(workspaceCom:GetDescendants()) do
        if IsMeuProp(obj) then
            AddProp(obj)
        end
    end
end

local function MonitorarNovosProps()
    local workspaceCom = workspace:FindFirstChild("WorkspaceCom")
    if not workspaceCom then return end

    local conn = workspaceCom.DescendantAdded:Connect(function(obj)
        if rgbAtivo and IsMeuProp(obj) then
            AddProp(obj)
        end
    end)

    table.insert(propConnections, conn)
end

TrollTab:AddToggle({
    Name = "Auto RGB Props",
    Default = false,
    Callback = function(state)

        rgbAtivo = state

        if rgbAtivo then

            ScanProps()
            MonitorarNovosProps()

            rgbConnection = RunService.Heartbeat:Connect(function()

                local hue = (tick() * 0.5) % 1
                local color = Color3.fromHSV(hue, 1, 1)

                for i = #meusProps, 1, -1 do
                    local prop = meusProps[i]

                    if not prop or not prop.Parent then
                        table.remove(meusProps, i)
                    else
                        pcall(function()
                            prop.ChangePropColor:InvokeServer(color)
                        end)
                    end
                end

            end)

        else

            if rgbConnection then
                rgbConnection:Disconnect()
                rgbConnection = nil
            end

            for _, c in ipairs(propConnections) do
                c:Disconnect()
            end

            propConnections = {}
            meusProps = {}

        end
    end
})

                local hum = LocalPlayer.Character:FindFirstChildOfClass("Humanoid")

local PROP_RAID = false
local Connection
local TargetTable = {}

local function GetMyProps()

    local props = {}
    local ws = workspace:FindFirstChild("WorkspaceCom")
    if not ws then return props end

    for _,folder in pairs(ws:GetChildren()) do
        for _,prop in pairs(folder:GetChildren()) do

            if prop:FindFirstChild("SetCurrentCFrame") and prop.Name:lower():find(LocalPlayer.Name:lower()) then
                table.insert(props,prop)
            end

        end
    end

    return props
end

local function GetTargets()

    local targets = {}

    for _,p in pairs(Players:GetPlayers()) do

        if p ~= LocalPlayer
        and p.Character
        and p.Character:FindFirstChild("HumanoidRootPart") then

            table.insert(targets,p)

        end

    end

    return targets
end

local function AssignTargets(props,targets)

    if #targets == 0 then return end

    for i,prop in pairs(props) do
        TargetTable[prop] = targets[((i-1) % #targets) + 1]
    end

end

local angle = 0
local height = 0
local direction = 1

local function StartRaid()

    if Connection then Connection:Disconnect() end

    Connection = RunService.Heartbeat:Connect(function()

        if not Character or Humanoid.Health <= 0 then return end

        local props = GetMyProps()
        local targets = GetTargets()

        if #props == 0 or #targets == 0 then return end

        AssignTargets(props,targets)

        angle += math.rad(15)

        if height > 4 then direction = -1 end
        if height < -4 then direction = 1 end

        height += 0.25 * direction

        for _,prop in pairs(props) do

            local target = TargetTable[prop]

            if target
            and target.Character
            and target.Character:FindFirstChild("HumanoidRootPart") then

                local root = target.Character.HumanoidRootPart

                local radius = math.random(2,5)

                local offset = Vector3.new(
                    math.cos(angle)*radius,
                    height,
                    math.sin(angle)*radius
                )

                pcall(function()

                    prop.SetCurrentCFrame:InvokeServer(
                        root.CFrame * CFrame.new(offset)
                    )

                end)

            end

        end

    end)

end

local function StopRaid()

    if Connection then
        Connection:Disconnect()
        Connection = nil
    end

    TargetTable = {}

end

local HAHA_ENABLED = false
local HAHA_CONNECTION

local TargetTable = {}
local SitTimer = {}
local DropTimer = {}

local CHANGE_TIME = 5
local DROP_Y = -503
local DROP_HOLD = 2

local t = 0

local function GetTargets()

    local targets = {}

    for _,p in pairs(Players:GetPlayers()) do

        if p ~= LocalPlayer
        and p.Character
        and p.Character:FindFirstChild("HumanoidRootPart")
        and p.Character:FindFirstChild("Humanoid") then

            table.insert(targets,p)

        end

    end

    return targets
end


local function AssignTargets(props)

    local targets = GetTargets()
    if #targets == 0 then return end

    for i,prop in ipairs(props) do

        if not TargetTable[prop] then
            TargetTable[prop] = targets[math.random(1,#targets)]
            SitTimer[prop] = tick()
        end

    end

end


function StartHaha()

if HAHA_CONNECTION then HAHA_CONNECTION:Disconnect() end

HAHA_CONNECTION = RunService.Heartbeat:Connect(function(dt)

    if not Character or not HRP then return end

    local props = GetMyProps()
    if #props == 0 then return end

    AssignTargets(props)

    t += dt * 6

    for i,prop in ipairs(props) do

        local target = TargetTable[prop]

        if not target
        or not target.Character
        or not target.Character:FindFirstChild("HumanoidRootPart")
        or not target.Character:FindFirstChild("Humanoid") then

            TargetTable[prop] = nil
            DropTimer[prop] = nil
            continue

        end

        local hum = target.Character.Humanoid
        local root = target.Character.HumanoidRootPart

        ------------------------------------------------
        -- SE SENTAR → DROP
        ------------------------------------------------

        if hum.Sit then

            if not DropTimer[prop] then
                DropTimer[prop] = tick()
            end

            local dropPos = Vector3.new(
                root.Position.X,
                DROP_Y,
                root.Position.Z
            )

            pcall(function()
                prop.SetCurrentCFrame:InvokeServer(
                    CFrame.new(dropPos)
                )
            end)

            -- segura 2 segundos lá
            if tick() - DropTimer[prop] < DROP_HOLD then
                continue
            end

            -- se morreu ou resetou troca alvo
            if hum.Health <= 0 then
                TargetTable[prop] = nil
                DropTimer[prop] = nil
            end

            continue
        end

        DropTimer[prop] = nil

        ------------------------------------------------
        -- MOVIMENTO NORMAL
        ------------------------------------------------

        local side = math.sin(t+i)*1.5
        local up = math.cos(t*1.3+i)*1.5
        local front = math.sin(t*0.7+i)*1

        local pos =
            root.Position
            + root.CFrame.RightVector * side
            + root.CFrame.LookVector * front
            + Vector3.new(0,up,0)

        local rot = CFrame.Angles(
            math.rad(t*200),
            math.rad(t*150),
            math.rad(t*100)
        )

        local cf = CFrame.new(pos) * rot

        pcall(function()

            prop.SetCurrentCFrame:InvokeServer(
                prop:GetPivot():Lerp(cf,0.35)
            )

        end)

        ------------------------------------------------
        -- TROCAR TARGET
        ------------------------------------------------

        if tick() - (SitTimer[prop] or 0) > CHANGE_TIME then

            local targets = GetTargets()

            if #targets > 0 then
                TargetTable[prop] = targets[math.random(1,#targets)]
                SitTimer[prop] = tick()
            end

        end

    end

end)

end


function StopHaha()

if HAHA_CONNECTION then
HAHA_CONNECTION:Disconnect()
HAHA_CONNECTION = nil
end

TargetTable = {}
SitTimer = {}
DropTimer = {}

end


TrollTab:AddToggle({
Name = "kill all",
Default = false,
Callback = function(v)

HAHA_ENABLED = v

if v then
StartHaha()
else
StopHaha()
end

end

local COBRA_ENABLED = false
local COBRA_CONNECTION

local FOLLOW_DISTANCE = 3.5
local SIDE_AMPLITUDE = 3.2
local UP_AMPLITUDE = 1.4
local WAVE_SPEED = 10
local PHASE_OFFSET = 1.1
local SMOOTHNESS = 0.35

function StartCobra()

if COBRA_CONNECTION then COBRA_CONNECTION:Disconnect() end

local time = 0

COBRA_CONNECTION = RunService.Heartbeat:Connect(function(dt)

    if not Character or not HRP or not Humanoid or Humanoid.Health <= 0 then return end

    local props = GetMyProps()
    if #props == 0 then return end

    time += dt * WAVE_SPEED

    local backDir = -HRP.CFrame.LookVector
    local rightDir = HRP.CFrame.RightVector
    local upDir = Vector3.new(0,1,0)

    for i, prop in ipairs(props) do

        local phase = time - (i * PHASE_OFFSET)

        -- movimento lateral mais suave
        local sideWave = math.sin(phase) * SIDE_AMPLITUDE

        -- movimento vertical mais natural
        local upWave = math.cos(phase * 0.8) * UP_AMPLITUDE

        -- pequeno segundo movimento para dar mais molejo
        local extraWave = math.sin(phase * 0.4) * 0.6

        local backOffset = backDir * (FOLLOW_DISTANCE * i)
        local sideOffset = rightDir * (sideWave + extraWave)
        local upOffset = upDir * upWave

        local pos = HRP.Position + backOffset + sideOffset + upOffset

        local cf = CFrame.new(pos, pos + backDir)

        pcall(function()
            prop.SetCurrentCFrame:InvokeServer(
                prop:GetPivot():Lerp(cf, SMOOTHNESS)
            )
        end)

    end

end)

end


function StopCobra()

if COBRA_CONNECTION then
COBRA_CONNECTION:Disconnect()
COBRA_CONNECTION = nil
end

end

local AIR_ENABLED = false
local AIR_CONNECTION

local SPEED = 1
local HEIGHT = 30
local SPACING = 12

local function StartAirSpread()

    if AIR_CONNECTION then AIR_CONNECTION:Disconnect() end

    local t = 0

    AIR_CONNECTION = RunService.Heartbeat:Connect(function(dt)

        if not Character or not HRP or not Humanoid or Humanoid.Health <= 0 then return end

        local props = GetMyProps() -- AQUI ESTÁ A CORREÇÃO
        if #props == 0 then return end

        t += dt * SPEED

        for i, prop in ipairs(props) do

            local height = HEIGHT + math.sin(t + i*1.2) * 4

            local angle = (i * 2.5)
            local radius = SPACING + (i * 2)

            local randomSide = math.sin(t*1.5 + i*3) * 4
            local randomFront = math.cos(t*1.3 + i*2) * 4

            local offset = Vector3.new(
                math.cos(angle) * radius + randomSide,
                height,
                math.sin(angle) * radius + randomFront
            )

            local pos = HRP.Position + offset

    if ARMS_CONNECTION then
        ARMS_CONNECTION:Disconnect()
    end

    ARMS_CONNECTION = RunService.Heartbeat:Connect(function()

        if not Character or not HRP or not Humanoid or Humanoid.Health <= 0 then return end

        local props = GetMyProps()
        if #props == 0 then return end

        local right = HRP.CFrame.RightVector
        local forward = HRP.CFrame.LookVector

        local count = #props
        local odd = count % 2 ~= 0

        local startIndex = 1

        -- PROP NA FRENTE (se número ímpar)
        if odd then

            local frontProp = props[1]

            local pos =
                HRP.Position
                + forward * frontDistance
                + Vector3.new(0,armsHeight,0)

            pcall(function()
                frontProp.SetCurrentCFrame:InvokeServer(
                    CFrame.new(pos)
                )
            end)

            startIndex = 2
        end

        -- PROPS NOS BRAÇOS (ASA CURVADA PARA TRÁS)
        local index = 0

        for i = startIndex, count do

            index += 1

            local side = (index % 2 == 0) and -1 or 1
            local step = math.ceil(index/2)

            local sideOffset = armsSpacing * step + 1.5
            local curveBack = step * wingCurve

            local pos =
                HRP.Position
                + right * (side * sideOffset)
                - forward * curveBack
                + Vector3.new(0,armsHeight,0)

            local prop = props[i]

            pcall(function()
                prop.SetCurrentCFrame:InvokeServer(
                    CFrame.new(pos)
                )
            end)

        end

    end)

end

local LINE_ENABLED = false
local LINE_CONNECTION

local FORWARD_DISTANCE = 2
local SPACING = 3
local SIDE_MOVE = 0.15       -- menor, bem suave
local WAVE_SPEED = 1.5        -- mais lento para suavidade
local HEIGHT_OFFSET = -2      -- mais para baixo, perto da perna
local SMOOTHNESS = 0.35

local function StartLine()

    if LINE_CONNECTION then LINE_CONNECTION:Disconnect() end

    local time = 0

    LINE_CONNECTION = RunService.Heartbeat:Connect(function(dt)

        if not Character or not HRP or not Humanoid or Humanoid.Health <= 0 then return end

        local props = GetMyProps()
        if #props == 0 then return end

        time += dt * WAVE_SPEED

        local forwardDir = HRP.CFrame.LookVector
        local rightDir = HRP.CFrame.RightVector

        local basePos =
            HRP.Position
            + forwardDir * FORWARD_DISTANCE
            + Vector3.new(0, HEIGHT_OFFSET, 0)

        for i,prop in ipairs(props) do

            local forwardOffset = forwardDir * (SPACING * (i-1))

            -- movimento lateral suave, último prop se mexe mais
            local dynamicSideMove = SIDE_MOVE * (0.8 + (i/#props) * 0.5)
            local sideWave = math.sin(time + i * 0.5) * dynamicSideMove
            local sideOffset = rightDir * sideWave

            local pos = basePos + forwardOffset + sideOffset

            local cf = CFrame.new(pos, pos + forwardDir)

            pcall(function()
                prop.SetCurrentCFrame:InvokeServer(
                    prop:GetPivot():Lerp(cf, SMOOTHNESS)
                )
            end)

        end

    end)

end


local function StopLine()

    if LINE_CONNECTION then
        LINE_CONNECTION:Disconnect()

    end)
